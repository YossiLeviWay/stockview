import React, { useState, useEffect } from 'react';

import { Timer, Trophy, SkipForward, Play, RefreshCcw, CheckCircle2, XCircle } from 'lucide-react';


const rawQuestions = [

  {

    id: 1,

    text: "השלימו את הסדרה: 1963, 1973, 1983...",

    options: ["1993, 2003, 2013", "1993, 1994, 1995", "2000, 2010, 2020", "1984, 1985, 1986"],

    correctAnswer: "1993, 2003, 2013"

  },

  {

    id: 2,

    text: "כתבו מספר הגדול מ-2000 המורכב מהספרות 0, 1, 2, 3 (פעם אחת בכל ספרה):",

    options: ["2013", "1320", "1023", "3210"],

    correctAnswer: "2013"

  },

  {

    id: 3,

    text: "איזה תרגיל פתרונו הוא 1?",

    options: ["22 + 1", "22 × 1", "22 - 22", "22 : 22"],

    correctAnswer: "22 : 22"

  },

  {

    id: 4,

    text: "השלימו: 500 - ____ = 360",

    options: ["140", "160", "240", "260"],

    correctAnswer: "140"

  },

  {

    id: 5,

    text: "פתרו: 845 + 87 =",

    options: ["932", "922", "832", "942"],

    correctAnswer: "932"

  },

  {

    id: 6,

    text: "פתרו: 6,000 - 780 =",

    options: ["5,220", "5,320", "4,220", "6,780"],

    correctAnswer: "5,220"

  },

  {

    id: 7,

    text: "פתרו: 25 × 8 =",

    options: ["200", "180", "250", "150"],

    correctAnswer: "200"

  },

  {

    id: 8,

    text: "פתרו: 40 : (4 + 6) =",

    options: ["4", "10", "16", "40"],

    correctAnswer: "4"

  },

  {

    id: 9,

    text: "בכיתה 28 תלמידים. המורה נתנה לכל תלמיד עיפרון ונשארו לה 12. כמה עפרונות היו לה בהתחלה?",

    options: ["40", "16", "30", "50"],

    correctAnswer: "40"

  },

  {

    id: 10,

    text: "מהו הסימן המתאים? 3 × 400 ____ 30 × 40",

    options: ["=", ">", "<"],

    correctAnswer: "="

  },

  {

    id: 11,

    text: "מהי השארית בתרגיל 28 : 5?",

    options: ["3", "2", "1", "אין שארית"],

    correctAnswer: "3"

  },

  {

    id: 12,

    text: "בבי״ס 100 תלמידי כיתות ד'. בד1 יש 32, בד2 יש 33. כמה לומדים בד3?",

    options: ["35", "34", "45", "25"],

    correctAnswer: "35"

  },

  {

    id: 13,

    text: "אם מחלקים 100 תלמידים לקבוצות של 10, כמה קבוצות יהיו?",

    options: ["10", "100", "1", "20"],

    correctAnswer: "10"

  },

  {

    id: 14,

    text: "איזו שעה תהיה רבע שעה אחרי 7:15, ועוד חצי שעה נסיעה?",

    options: ["8:00", "7:45", "8:15", "7:30"],

    correctAnswer: "8:00"

  }

];


const App = () => {

  const [gameState, setGameState] = useState('START'); // START, PLAYING, END

  const [questions, setQuestions] = useState([]);

  const [currentIndex, setCurrentIndex] = useState(0);

  const [score, setScore] = useState(0);

  const [timeLeft, setTimeLeft] = useState(120);

  const [feedback, setFeedback] = useState(null);

  const [userResponses, setUserResponses] = useState({});


  const shuffle = (array) => {

    return [...array].sort(() => Math.random() - 0.5);

  };


  const startGame = () => {

    // 1. Shuffle the question list

    const shuffledQuestions = shuffle(rawQuestions);

    

    // 2. For each question, shuffle its options as well

    const questionsWithShuffledOptions = shuffledQuestions.map(q => ({

      ...q,

      options: shuffle(q.options)

    }));


    setQuestions(questionsWithShuffledOptions);

    setCurrentIndex(0);

    setScore(0);

    setTimeLeft(120);

    setUserResponses({});

    setGameState('PLAYING');

    setFeedback(null);

  };


  useEffect(() => {

    let timer;

    if (gameState === 'PLAYING' && timeLeft > 0) {

      timer = setInterval(() => {

        setTimeLeft((prev) => prev - 1);

      }, 1000);

    } else if (timeLeft === 0 && gameState === 'PLAYING') {

      setGameState('END');

    }

    return () => clearInterval(timer);

  }, [gameState, timeLeft]);


  const handleAnswer = (answer) => {

    if (feedback) return;


    const currentQ = questions[currentIndex];

    const isCorrect = answer === currentQ.correctAnswer;

    

    setUserResponses(prev => ({

      ...prev,

      [currentIndex]: answer

    }));


    if (isCorrect) {

      setScore(prev => prev + 1);

      setFeedback({ type: 'success', message: 'כל הכבוד! תשובה נכונה' });

    } else {

      setFeedback({ type: 'error', message: `לא נכון. התשובה הנכונה היא: ${currentQ.correctAnswer}` });

    }


    setTimeout(() => {

      setFeedback(null);

      if (currentIndex < questions.length - 1) {

        setCurrentIndex(prev => prev + 1);

      } else {

        setGameState('END');

      }

    }, 1500);

  };


  const skipQuestion = () => {

    setUserResponses(prev => ({

      ...prev,

      [currentIndex]: 'SKIPPED'

    }));

    

    setFeedback(null);

    if (currentIndex < questions.length - 1) {

      setCurrentIndex(prev => prev + 1);

    } else {

      setGameState('END');

    }

  };


  const formatTime = (seconds) => {

    const mins = Math.floor(seconds / 60);

    const secs = seconds % 60;

    return `${mins}:${secs.toString().padStart(2, '0')}`;

  };


  return (

    <div className="min-h-screen bg-slate-50 font-sans text-slate-900 flex flex-col items-center p-4 md:p-8" dir="rtl">

      {/* Header Area */}

      <header className="w-full max-w-3xl bg-white rounded-2xl shadow-sm p-4 mb-8 flex justify-between items-center sticky top-4 z-10 border border-slate-100">

        <div className="flex items-center gap-3">

          <div className="bg-blue-100 p-2 rounded-lg text-blue-600">

            <Trophy size={24} />

          </div>

          <div>

            <p className="text-xs text-slate-500 font-medium">הניקוד שלך</p>

            <p className="text-xl font-bold">{score}</p>

          </div>

        </div>


        <div className="flex items-center gap-3">

          <div className={`p-2 rounded-lg ${timeLeft < 20 && gameState === 'PLAYING' ? 'bg-red-100 text-red-600 animate-pulse' : 'bg-green-100 text-green-600'}`}>

            <Timer size={24} />

          </div>

          <div>

            <p className="text-xs text-slate-500 font-medium">זמן נותר</p>

            <p className={`text-xl font-bold ${timeLeft < 20 && gameState === 'PLAYING' ? 'text-red-600' : ''}`}>

              {formatTime(timeLeft)}

            </p>

          </div>

        </div>

      </header>


      {/* Main Game Container */}

      <main className="w-full max-w-3xl flex-grow flex flex-col">

        {gameState === 'START' && (

          <div className="bg-white rounded-3xl shadow-xl p-8 text-center flex-grow flex flex-col justify-center items-center gap-6 border-b-8 border-blue-500 transition-all">

            <h1 className="text-4xl font-extrabold text-slate-800 tracking-tight">אתגר המתמטיקה - כיתה ד'</h1>

            <p className="text-lg text-slate-600 max-w-md font-medium">

              התשובות מעורבבות בכל פעם! ענו על כמה שיותר שאלות ב-2 דקות.

            </p>

            <button 

              onClick={startGame}

              className="group flex items-center gap-2 bg-blue-600 hover:bg-blue-700 text-white px-10 py-5 rounded-full text-2xl font-bold transition-all transform hover:scale-105 active:scale-95 shadow-lg shadow-blue-200"

            >

              בואו נתחיל!

              <Play className="fill-current ml-2" />

            </button>

          </div>

        )}


        {gameState === 'PLAYING' && questions.length > 0 && (

          <div className="relative flex-grow flex flex-col">

            <div className="bg-white rounded-3xl shadow-xl p-8 border-b-8 border-blue-200 transition-all">

              <div className="mb-8">

                <span className="bg-blue-50 text-blue-600 px-3 py-1 rounded-full text-sm font-bold">

                  שאלה {currentIndex + 1} מתוך {questions.length}

                </span>

                <h2 className="text-2xl font-bold mt-4 leading-relaxed text-slate-800">

                  {questions[currentIndex].text}

                </h2>

              </div>


              <div className="grid grid-cols-1 md:grid-cols-2 gap-4 mb-8">

                {questions[currentIndex].options.map((option, idx) => (

                  <button

                    key={idx}

                    disabled={feedback !== null}

                    onClick={() => handleAnswer(option)}

                    className={`text-right p-5 rounded-2xl border-2 transition-all font-semibold text-lg

                      ${feedback ? 'opacity-50 cursor-not-allowed' : 'hover:border-blue-500 hover:bg-blue-50 active:bg-blue-100 border-slate-100'}`}

                  >

                    {option}

                  </button>

                ))}

              </div>


              <div className="flex justify-between items-center pt-6 border-t border-slate-100">

                <button 

                  onClick={skipQuestion}

                  disabled={feedback !== null}

                  className="flex items-center gap-2 text-slate-400 hover:text-slate-600 font-bold px-4 py-2 transition-colors"

                >

                  <SkipForward size={20} />

                  דלג על השאלה

                </button>

              </div>

            </div>


            {feedback && (

              <div className={`mt-6 p-5 rounded-2xl flex items-center gap-3 animate-in fade-in slide-in-from-bottom-4 duration-300 shadow-md

                ${feedback.type === 'success' ? 'bg-green-100 text-green-800 border-r-4 border-green-500' : 'bg-red-100 text-red-800 border-r-4 border-red-500'}`}>

                {feedback.type === 'success' ? <CheckCircle2 className="shrink-0" /> : <XCircle className="shrink-0" />}

                <span className="font-bold text-lg">{feedback.message}</span>

              </div>

            )}

          </div>

        )}


        {gameState === 'END' && (

          <div className="space-y-6 pb-12">

            <div className="bg-white rounded-3xl shadow-xl p-10 text-center flex flex-col justify-center items-center gap-6 border-b-8 border-green-500">

              <div className="bg-yellow-100 p-5 rounded-full text-yellow-500 mb-2">

                <Trophy size={48} />

              </div>

              <div>

                <h2 className="text-4xl font-extrabold text-slate-800 mb-2">כל הכבוד!</h2>

                <p className="text-2xl text-slate-600 font-medium">צברת {score} נקודות מתוך {Object.keys(userResponses).length} שאלות שענית עליהן</p>

              </div>

              <button 

                onClick={startGame}

                className="flex items-center gap-2 bg-blue-600 hover:bg-blue-700 text-white px-8 py-4 rounded-2xl text-xl font-bold transition-all shadow-lg"

              >

                <RefreshCcw size={20} />

                נסה שוב

              </button>

            </div>


            <div className="mt-8">

              <h3 className="text-2xl font-bold text-slate-800 mb-6 px-2 underline decoration-blue-500 decoration-4 underline-offset-8">סיכום התשובות שלך:</h3>

              <div className="grid gap-6">

                {questions.map((q, idx) => {

                  const userAns = userResponses[idx];

                  if (!userAns) return null;


                  return (

                    <div key={idx} className="bg-white rounded-2xl shadow-sm p-6 border border-slate-100">

                      <div className="flex justify-between items-start mb-4">

                        <h4 className="font-bold text-lg text-slate-800 max-w-[85%]">

                          {idx + 1}. {q.text}

                        </h4>

                        {userAns === q.correctAnswer ? (

                          <span className="bg-green-100 text-green-600 p-1 rounded-full"><CheckCircle2 size={20} /></span>

                        ) : (

                          <span className="bg-red-100 text-red-600 p-1 rounded-full"><XCircle size={20} /></span>

                        )}

                      </div>

                      

                      <div className="grid grid-cols-1 sm:grid-cols-2 gap-3">

                        {q.options.map((opt, optIdx) => {

                          const isCorrect = opt === q.correctAnswer;

                          const isSelected = opt === userAns;

                          

                          let bgClass = "bg-slate-50 text-slate-600 border-slate-100";

                          if (isCorrect) bgClass = "bg-green-50 border-green-500 text-green-700 border-2";

                          if (isSelected && !isCorrect) bgClass = "bg-red-50 border-red-500 text-red-700 border-2";


                          return (

                            <div key={optIdx} className={`p-3 rounded-xl border text-sm font-bold flex justify-between items-center ${bgClass}`}>

                              {opt}

                              {isSelected && <span className="text-[10px] uppercase px-1.5 py-0.5 bg-white/50 rounded ml-2">הבחירה שלך</span>}

                            </div>

                          );

                        })}

                      </div>

                      

                      {userAns === 'SKIPPED' && (

                        <div className="mt-3 text-sm font-bold text-orange-500 bg-orange-50 p-2 rounded-lg inline-block">

                          דילגת על שאלה זו

                        </div>

                      )}

                    </div>

                  );

                })}

              </div>

            </div>

          </div>

        )}

      </main>


      <footer className="mt-auto pt-8 text-slate-400 text-sm font-medium">

        מערכת תרגול משרד החינוך - כיתה ד'

      </footer>

    </div>

  );

};


export default App;
